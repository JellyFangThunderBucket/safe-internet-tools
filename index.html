🛡 Mom Safety Filter

Adds a big warning when a website uses scary scam words like “dark web,” “virus detected,” or “immediate action required.”

Install / Download
Show script code 
// ==UserScript==
// @name         Mom Safety Filter - Scam Warning
// @namespace    mom-safety-filter
// @version      1.0
// @description  Warns before scammy scare pages trick someone into spending money
// @match        *://*/*
// @grant        none
// ==/UserScript==

(function () {
  const dangerWords = [
    "dark web",
    "your information has been leaked",
    "your device is infected",
    "virus detected",
    "trojan detected",
    "click here to remove",
    "your ip address is exposed",
    "vpn required",
    "renew your protection",
    "security alert",
    "immediate action required",
    "your computer is at risk",
    "your phone is infected",
    "call microsoft",
    "call apple support"
  ];

  function scanPage() {
    if (!document.body) return;
    const text = document.body.innerText.toLowerCase();
    const found = dangerWords.some(word => text.includes(word));

    if (found && !document.getElementById("momSafetyWarning")) {
      const warning = document.createElement("div");
      warning.id = "momSafetyWarning";
      warning.innerHTML = `
        Pause. This may be a scam.

        Do not click buttons, call numbers, or pay money from this page.

        Close this tab and ask Cody first.
      `;

      warning.style.position = "fixed";
      warning.style.top = "0";
      warning.style.left = "0";
      warning.style.right = "0";
      warning.style.zIndex = "999999999";
      warning.style.background = "#fff3cd";
      warning.style.color = "#3b2f00";
      warning.style.fontSize = "20px";
      warning.style.fontFamily = "Arial, sans-serif";
      warning.style.padding = "18px";
      warning.style.textAlign = "center";
      warning.style.borderBottom = "4px solid #d39e00";
      warning.style.boxShadow = "0 4px 12px rgba(0,0,0,0.3)";

      document.body.appendChild(warning);
    }
  }

  scanPage();
  setInterval(scanPage, 2000);
})();

📖 Cozy Reading Mode

Adds a friendly button that makes websites softer, bigger, and easier to read.

Install / Download
Show script code 
// ==UserScript==
// @name         Mom's Cozy Reading Mode
// @namespace    mom-cozy-reading
// @version      1.0
// @description  Makes websites softer, bigger, and easier to read
// @match        *://*/*
// @grant        none
// ==/UserScript==

(function () {
  const button = document.createElement("button");

  button.textContent = "Aa";
  button.style.position = "fixed";
  button.style.bottom = "20px";
  button.style.right = "20px";
  button.style.zIndex = "999999";
  button.style.fontSize = "20px";
  button.style.padding = "12px 15px";
  button.style.borderRadius = "999px";
  button.style.border = "none";
  button.style.background = "#f3e7d3";
  button.style.color = "#2b2118";
  button.style.boxShadow = "0 4px 12px rgba(0,0,0,0.25)";

  let cozy = false;

  button.onclick = function () {
    cozy = !cozy;

    document.body.style.background = cozy ? "#f8f1e7" : "";
    document.body.style.color = cozy ? "#2b2118" : "";
    document.body.style.fontSize = cozy ? "20px" : "";
    document.body.style.lineHeight = cozy ? "1.7" : "";

    document.querySelectorAll("p, li, span, div, article").forEach(el => {
      el.style.fontSize = cozy ? "20px" : "";
      el.style.lineHeight = cozy ? "1.7" : "";
    });

    button.textContent = cozy ? "✓" : "Aa";
  };

  document.body.appendChild(button);
})();
